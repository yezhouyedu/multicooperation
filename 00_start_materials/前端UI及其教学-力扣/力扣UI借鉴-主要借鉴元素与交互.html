1722. 执行交换操作后的最小汉明距离
中等
相关标签
相关企业
提示

给你两个整数数组 source 和 target ，长度都是 n 。还有一个数组 allowedSwaps ，其中每个 allowedSwaps[i] = [ai, bi] 表示你可以交换数组 source 中下标为 ai 和 bi（下标从 0 开始）的两个元素。注意，你可以按 任意 顺序 多次 交换一对特定下标指向的元素。

相同长度的两个数组 source 和 target 间的 汉明距离 是元素不同的下标数量。形式上，其值等于满足 source[i] != target[i] （下标从 0 开始）的下标 i（0 <= i <= n-1）的数量。

在对数组 source 执行 任意 数量的交换操作后，返回 source 和 target 间的 最小汉明距离 。

 

示例 1：

输入：source = [1,2,3,4], target = [2,1,4,5], allowedSwaps = [[0,1],[2,3]]
输出：1
解释：source 可以按下述方式转换：
- 交换下标 0 和 1 指向的元素：source = [2,1,3,4]
- 交换下标 2 和 3 指向的元素：source = [2,1,4,3]
source 和 target 间的汉明距离是 1 ，二者有 1 处元素不同，在下标 3 。

示例 2：

输入：source = [1,2,3,4], target = [1,3,2,4], allowedSwaps = []
输出：2
解释：不能对 source 执行交换操作。
source 和 target 间的汉明距离是 2 ，二者有 2 处元素不同，在下标 1 和下标 2 。

示例 3：

输入：source = [5,1,2,4,3], target = [1,5,4,2,3], allowedSwaps = [[0,4],[4,2],[1,3],[1,4]]
输出：0

 

提示：

  • n == source.length == target.length
  • 1 <= n <= 105
  • 1 <= source[i], target[i] <= 105
  • 0 <= allowedSwaps.length <= 105
  • allowedSwaps[i].length == 2
  • 0 <= ai, bi <= n - 1
  • ai != bi

这题的关键点是证明同一个连通块内的元素可以任意重排，我来一个简易证明。从graph的角度看，一个连通块，要么是无环的tree结构，要么是有环的结构。先证明前者的情况，使用归纳法，假设一棵有k个nodes的tree可以实现nodes的任意重排，当tree有k+1个nodes时，任意找一个leaf node先完成，然后去掉这个完成的leaf node，剩下的结构仍然是tree结构，且只有k个nodes，则直接利用归纳假设证明完毕。再看有环的情况，此时一定可以通过去掉一定数量的edge，将环消除，形成一个连通树，这就可以使用上文的结论直接得证。

//并查集操作
int fa[100001];
void initFa(int s){iota(fa,fa+s,0);}
int  gf(int i) {return fa[i]==i?i:fa[i]=gf(fa[i]);}
void mer(int i, int j){fa[gf(i)] = gf(j);}

//求两个已排序集合的交集元素数量
constexpr auto itsCnt = [](auto &l, auto &r){
    auto i1=l.cbegin(),i2=r.cbegin(),e1=l.cend(),e2=r.cend();
    int re=0;
    while(i1!=e1 && i2!=e2)
        tie(re,i1,i2)=tuple(re+(*i1==*i2), i1+(*i1<=*i2), i2+(*i2<=*i1));
    return re;
};
 //本题的海明距离       
constexpr auto dis = [](auto &l, auto &r){return l.size()-itsCnt(l,r);};

class Solution {
public:
    int minimumHammingDistance(vector<int>& source, vector<int>& target, vector<vector<int>>& allowedSwaps) {        
        auto getG = [&](){
            using RG = ranges::iota_view<int,int>;
            constexpr auto do_each = ranges::for_each; 
            
            initFa(source.size());
            do_each(allowedSwaps,[](auto &v){mer(v[0],v[1]);});
            unordered_map<int,vector<int>> gr;
            do_each(RG(0,source.size()),[&](auto i){gr[gf(i)].push_back(i);});
            return gr;
        }; 
        auto toV = [ ](auto &i_v, auto &tr)->vector<int>{
            auto v = i_v.second | views::transform(tr);
            vector<int> re(v.begin(),v.end());
            ranges::sort(re);
            return re;           
        }; 
        auto toVs= [&](auto &gr, auto &&tr)->vector<vector<int>>{
            auto v = gr|views::transform([&](auto &iv){return toV(iv,tr);});
            return {v.begin(),v.end()};
        };

        //开始干活 
        auto const gr = getG();
        auto const sg = toVs(gr,[&](int i){return source[i];});
        auto const tg = toVs(gr,[&](int i){return target[i];});
        return inner_product(sg.begin(),sg.end(),tg.begin(),0,std::plus<int>(),dis);
    }
};

class Solution {
public:
class UnionFind{
public:
vector parent;
int cnt;
UnionFind(int n){
parent.resize(n);
cnt = n;
for(int i=0;i<n;i++){
parent[i] = i;
}
}
int find(int x){
if(parent[x] != x){
parent[x] = find(parent[x]);
}
return parent[x];
}
bool Unite(int x, int y){
int rootx = find(x);
int rooty = find(y);
if(rootx == rooty) return false;
else{
parent[rootx] = rooty;
cnt--;
return true;
}
}
};
int minimumHammingDistance(vector& source, vector& target, vector<vector>& allowedSwaps) {
int n = source.size();
int m = allowedSwaps.size();
int ans = 0;
UnionFind u(n);
for(int i=0;i<m;i++){
u.Unite(allowedSwaps[i][0], allowedSwaps[i][1]);
}
unordered_map<int, vector> check;
for(int i=0;i<n;i++){
int father = u.find(i);
check[father].push_back(i);
}
for(const auto& curr : check){
unordered_map<int, int> cmp;
const vector& index = curr.second;
for(int i=0;i<index.size();i++){
cmp[source[index[i]]]++;
}
int tempcnt = 0;
for(int i=0;i<index.size();i++){
if(!cmp.count(target[index[i]])){
tempcnt++;
}
else if(cmp[target[index[i]]] == 0){
tempcnt++;
}
else cmp[target[index[i]]]--;
}
ans += tempcnt;
}
return ans;
}
};

class Solution {
public:
    static const int N = 1e5 + 10;
    int p[N];

    int find(int x) {
        if (p[x] != x) p[x] = find(p[x]);
        return p[x];
    }

    int minimumHammingDistance(vector<int>& s, vector<int>& t, vector<vector<int>>& allow) {
        int n = s.size();
        for (int i = 0; i < n; i++) p[i] = i;
        for (auto &v: allow) {
            int a = v[0], b = v[1];
            int pa = find(a), pb = find(b);
            p[pb] = pa;
        }
        unordered_map<int, unordered_map<int, int>> cnt;
        for (int i = 0; i < n; i++) {
            int pa = find(i);
            cnt[pa][s[i]]++;
        }
        int res = n;
        for (int i = 0; i < n; i++) {
            int a = s[i], b = t[i];
            if (cnt[find(i)][b]) {
                res--;
                cnt[find(i)][b]--;
            }            
        }
        return res;
    }
};
